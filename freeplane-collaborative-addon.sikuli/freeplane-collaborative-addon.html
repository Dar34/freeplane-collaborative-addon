def doAction(action):
    # compare : wait("compare-button.png")
    # update : wait("update-button.png")
    # commit : wait("commit-button.png")
    #click(action + "-button.png")
    #return
    if action == "compare":
        type(Key.F10)
    elif action == "update":
        type(Key.F11)
    elif action == "commit":
        type(Key.F12)
    else:
        popup("Unknown action")
